### Network Performance

#### Request Optimization

```javascript
// Batch API requests
class ApiBatcher {
  constructor() {
    this.queue = [];
    this.timeout = null;
  }
  
  addRequest(request) {
    return new Promise((resolve, reject) => {
      this.queue.push({ request, resolve, reject });
      this.scheduleBatch();
    });
  }
  
  scheduleBatch() {
    if (this.timeout) return;
    
    this.timeout = setTimeout(() => {
      this.processBatch();
      this.timeout = null;
    }, 100); // Batch every 100ms
  }
  
  async processBatch() {
    const batch = this.queue.splice(0);
    if (batch.length === 0) return;
    
    try {
      const responses = await this.executeBatch(batch.map(b => b.request));
      batch.forEach((item, index) => {
        item.resolve(responses[index]);
      });
    } catch (error) {
      batch.forEach(item => item.reject(error));
    }
  }
  
  async executeBatch(requests) {
    // Implement batch API call
    return Promise.all(requests.map(req => fetch(req)));
  }
}

const apiBatcher = new ApiBatcher();

// Usage
const data1 = await apiBatcher.addRequest('/api/data1');
const data2 = await apiBatcher.addRequest('/api/data2');
```

#### Caching Strategy

```javascript
// Implement intelligent caching
class CacheManager {
  constructor() {
    this.cache = new Map();
    this.maxSize = 100;
  }
  
  get(key) {
    const item = this.cache.get(key);
    if (!item) return null;
    
    // Check if item is expired
    if (Date.now() > item.expiresAt) {
      this.cache.delete(key);
      return null;
    }
    
    // Move to end (LRU)
    this.cache.delete(key);
    this.cache.set(key, item);
    return item.data;
  }
  
  set(key, data, ttl = 300000) { // 5 minutes default TTL
    // Remove oldest item if cache is full
    if (this.cache.size >= this.maxSize) {
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }
    
    this.cache.set(key, {
      data,
      expiresAt: Date.now() + ttl,
      createdAt: Date.now()
    });
  }
  
  clear() {
    this.cache.clear();
  }
}

const cache = new CacheManager();

// Usage
async function fetchData(url) {
  const cached = cache.get(url);
  if (cached) return cached;
  
  const response = await fetch(url);
  const data = await response.json();
  cache.set(url, data);
  return data;
}
```

```rust
use tokio::time::{sleep, Duration};
use futures::stream::{StreamExt, FuturesUnordered};

pub async fn process_files(files: Vec<String>) -> Result<Vec<String>, Box<dyn std::error::Error>> {
    let mut futures = FuturesUnordered::new();
    
    // Process files concurrently with a limit
    for file in files {
        if futures.len() < 10 { // Limit to 10 concurrent operations
            futures.push(tokio::spawn(process_single_file(file)));
        } else {
            // Wait for one to complete before adding more
            if let Some(result) = futures.next().await {
                result?;
            }
            futures.push(tokio::spawn(process_single_file(file)));
        }
    }
    
    // Collect remaining results
    let mut results = Vec::new();
    while let Some(result) = futures.next().await {
        results.push(result?);
    }
    
    Ok(results)
}
```

#### Memory Pool

```rust
use std::sync::Arc;
use std::collections::VecDeque;

pub struct BufferPool {
    pool: Arc<std::sync::Mutex<VecDeque<Vec<u8>>>,
    buffer_size: usize,
}

impl BufferPool {
    pub fn new(buffer_size: usize, pool_size: usize) -> Self {
        let pool = Arc::new(std::sync::Mutex::new(VecDeque::with_capacity(pool_size)));
        
        // Pre-allocate buffers
        let mut pool_guard = pool.lock().unwrap();
        for _ in 0..pool_size {
            pool_guard.push_back(vec![0u8; buffer_size]);
        }
        
        Self { pool, buffer_size }
    }
    
    pub fn get_buffer(&self) -> Vec<u8> {
        let mut pool = self.pool.lock().unwrap();
        if let Some(mut buffer) = pool.pop_front() {
            buffer.clear();
            buffer
        } else {
            vec![0u8; self.buffer_size]
        }
    }
    
    pub fn return_buffer(&self, buffer: Vec<u8>) {
        let mut pool = self.pool.lock().unwrap();
        if buffer.len() == self.buffer_size {
            pool.push_front(buffer);
        }
    }
}
```

### Database Optimization

#### Connection Pooling

```rust
use sqlx::postgres::PgPoolOptions;
use std::time::Duration;

pub struct DatabasePool {
    pool: sqlx::PgPool,
}

impl DatabasePool {
    pub async fn new(database_url: &str) -> Result<Self, Box<dyn std::error::Error>> {
        let pool = PgPoolOptions::new()
            .max_connections(20)
            .min_connections(5)
            .acquire_timeout(Duration::from_secs(30))
            .idle_timeout(Duration::from_secs(600))
            .max_lifetime(Duration::from_secs(1800))
            .connect(database_url)
            .await?;
        
        Ok(Self { pool })
    }
    
    pub async fn execute_query(&self, query: &str) -> Result<sqlx::PgQueryResult, sqlx::Error> {
        sqlx::query(query).execute(&self.pool).await
    }
    
    pub async fn fetch_user(&self, user_id: i64) -> Result<Option<User>, sqlx::Error> {
        sqlx::query_as!(
            "SELECT id, name, email FROM users WHERE id = $1",
            user_id
        )
        .fetch_optional(&self.pool)
        .await
    }
}
```

#### Query Optimization

```rust
// Use prepared statements for repeated queries
pub struct UserRepository {
    pool: sqlx::PgPool,
}

impl UserRepository {
    pub fn new(pool: sqlx::PgPool) -> Self {
        Self { pool }
    }
    
    // Prepared statement for better performance
    pub async fn find_by_email(&self, email: &str) -> Result<Option<User>, sqlx::Error> {
        sqlx::query_as!(
            "SELECT id, name, email FROM users WHERE email = $1",
            email
        )
        .fetch_optional(&self.pool)
        .await
    }
    
    // Batch operations
    pub async fn create_users(&self, users: Vec<NewUser>) -> Result<Vec<User>, sqlx::Error> {
        let mut transaction = self.pool.begin().await?;
        
        let mut results = Vec::new();
        for user in users {
            let result = sqlx::query_as!(
                "INSERT INTO users (name, email) VALUES ($1, $2) RETURNING id, name, email",
                user.name,
                user.email
            )
            .fetch_one(&mut *transaction)
            .await?;
            
            results.push(result);
        }
        
        transaction.commit().await?;
        Ok(results)
    }
}
```

#### Image Optimization

```jsx
import { useState, useRef, useEffect } from 'react';

const LazyImage = ({ src, alt, className }) => {
  const [isLoaded, setIsLoaded] = useState(false);
  const [isInView, setIsInView] = useState(false);
  const imgRef = useRef();
  
  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setIsInView(true);
          observer.disconnect();
        }
      },
      { threshold: 0.1 }

    if (imgRef.current) {
      observer.observe(imgRef.current);
    }
    
    return () => observer.disconnect();
  }, []);
  
  return (
    <div ref={imgRef} className={className}>
      {isInView && (
        <img
          src={src}
          alt={alt}
          loading="lazy"
          onLoad={() => setIsLoaded(true)}
          style={{ opacity: isLoaded ? 1 : 0 }}
        />
      )}
    </div>
  );
};
```

## Performance Monitoring

### Metrics Collection

```javascript
// Performance monitoring
class PerformanceMonitor {
  constructor() {
    this.metrics = {
      startup: 0,
      render: 0,
      network: 0,
      memory: 0
    };
    this.observers = [];
  }
  
  startMonitoring() {
    // Monitor startup time
    const startTime = performance.now();
    
    window.addEventListener('load', () => {
      this.metrics.startup = performance.now() - startTime;
      this.reportMetrics();
    });
    
    // Monitor render performance
    this.observeRenderPerformance();
    
    // Monitor network performance
    this.observeNetworkPerformance();
    
    // Monitor memory usage
    this.observeMemoryUsage();
  }
  
  observeRenderPerformance() {
    const observer = new PerformanceObserver((list) => {
      for (const entry of list.getEntries()) {
        if (entry.entryType === 'measure') {
          this.metrics.render += entry.duration;
        }
      }
    });
    
    observer.observe({ entryTypes: ['measure'] });
  }
  
  observeNetworkPerformance() {
    const observer = new PerformanceObserver((list) => {
      for (const entry of list.getEntries()) {
        if (entry.entryType === 'resource') {
          this.metrics.network += entry.duration;
        }
      }
    });
    
    observer.observe({ entryTypes: ['resource'] });
  }
  
  observeMemoryUsage() {
    const checkMemory = () => {
      if (performance.memory) {
        this.metrics.memory = performance.memory.usedJSHeapSize;
      }
    };
    
    setInterval(checkMemory, 5000); // Check every 5 seconds
  }
  
  reportMetrics() {
    console.log('Performance Metrics:', this.metrics);
    
    // Send to analytics service
    if (navigator.sendBeacon) {
      const data = JSON.stringify(this.metrics);
      navigator.sendBeacon('/api/metrics', data);
    }
  }
}

// Initialize monitoring
const monitor = new PerformanceMonitor();
monitor.startMonitoring();
```

### Performance Budget

```javascript
// Define performance budget
const performanceBudget = {
  startup: 3000,        // 3 seconds
  render: 100,          // 100ms per frame
  network: 500,         // 500ms total
  memory: 100 * 1024 * 1024  // 100MB
};

// Check budget compliance
function checkPerformanceBudget(metrics) {
  const violations = [];
  
  if (metrics.startup > performanceBudget.startup) {
    violations.push(`Startup time exceeded: ${metrics.startup}ms`);
  }
  
  if (metrics.render > performanceBudget.render) {
    violations.push(`Render time exceeded: ${metrics.render}ms`);
  }
  
  if (metrics.network > performanceBudget.network) {
    violations.push(`Network time exceeded: ${metrics.network}ms`);
  }
  
  if (metrics.memory > performanceBudget.memory) {
    violations.push(`Memory usage exceeded: ${metrics.memory / 1024 / 1024}MB`);
  }
  
  return violations;
}
```

```javascript
// CPU profiling
function startCPUProfile() {
  if (window.performance && window.performance.mark) {
    window.performance.mark('profile-start');
  }
}

function stopCPUProfile() {
  if (window.performance && window.performance.mark) {
    window.performance.mark('profile-end');
    window.performance.measure('profile', 'profile-start', 'profile-end');
    
    const measures = window.performance.getEntriesByName('profile');
    console.log('CPU Profile:', measures);
  }
}

// Memory profiling
function getMemoryProfile() {
  if (performance.memory) {
    return {
      used: performance.memory.usedJSHeapSize,
      total: performance.memory.totalJSHeapSize,
      limit: performance.memory.jsHeapSizeLimit
    };
  }
  return null;
}
```

## Best Practices

1. **Measure First**: Always measure before optimizing
2. **Focus on Critical Path**: Optimize what users see first
3. **Use Caching**: Implement intelligent caching strategies
4. **Lazy Load**: Load resources only when needed
5. **Monitor Continuously**: Keep track of performance over time
6. **Set Budgets**: Define and enforce performance budgets
7. **Test Regularly**: Test performance on real devices
8. **Stay Updated**: Keep dependencies and tools updated
